# Hybrid Git Runtime Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add first-launch Git availability checking and a private Commit+ managed Git runtime fallback when no valid system Git is installed.

**Architecture:** Add a focused `GitRuntimeManager` that resolves system Git first, then an app-managed runtime under Application Support, then drives installation through a pinned app-owned manifest. Keep the existing `GitStatusService` command surface intact by replacing only the executable lookup. Gate `ContentView` behind runtime readiness so users see Git setup before opening or cloning a repository.

**Tech Stack:** Swift, SwiftUI, `Process`, `URLSession`, CryptoKit SHA-256, XCTest, `xcodebuild`, app-owned GitHub release artifact.

---

## File Structure

- Create `macgit/Services/GitRuntimeModels.swift`: runtime source, availability, manifest, and error types.
- Create `macgit/Services/GeneratedGitRuntimeManifest.swift`: generated manifest values for the app-owned runtime artifact.
- Create `macgit/Services/GitRuntimeManager.swift`: actor that resolves, validates, downloads, verifies, extracts, and returns the active Git executable URL.
- Create `macgit/Views/MainWindow/GitRuntimeSetupView.swift`: first-launch setup UI shown before the repo picker when Git is missing or installation fails.
- Modify `macgit/Services/GitStatusService.swift`: replace `gitExecutable()` with async runtime resolution.
- Modify `macgit/Services/GitStatusService+Stage.swift`: make patch application use the same runtime resolver.
- Modify `macgit/Views/MainWindow/ContentView.swift`: add the first-launch runtime gate.
- Create `macgitTests/GitRuntimeManagerTests.swift`: unit tests for source priority, validation, checksum behavior, and managed install state.
- Create `scripts/package-managed-git-runtime.sh`: release helper that packages a prebuilt private Git runtime and emits the manifest values the app pins.

## External Runtime Source Decision

Do not download directly from SourceForge or ask users to install Xcode Command Line Tools as the primary fallback. The macOS page on `git-scm.com` points to Apple Command Line Tools, Homebrew, or MacPorts, all of which install globally. The old `git-osx-installer` SourceForge project is marked abandoned. Commit+ should use an app-owned release artifact such as:

```text
https://github.com/Tranthanh98/macgit/releases/download/git-runtime-2.45.2-commitplus.1/git-runtime-2.45.2-commitplus.1-macos-universal.tar.gz
```

The exact checksum is generated by `scripts/package-managed-git-runtime.sh`, which writes `macgit/Services/GeneratedGitRuntimeManifest.swift` so the app pins the URL, version, checksum, and archive size from the same packaged artifact.

```swift
import Foundation
import CryptoKit

protocol GitRuntimeProcessRunning: AnyObject {
    func run(executableURL: URL, arguments: [String], currentDirectoryURL: URL?) async throws -> String
}

protocol GitRuntimeDownloading: AnyObject {
    func download(from url: URL) async throws -> Data
}

protocol GitRuntimeExtracting: AnyObject {
    func extract(archiveURL: URL, to destinationURL: URL) throws
}

struct GitRuntimeConfiguration {
    let applicationSupportDirectory: URL
    let candidateSystemGitURLs: [URL]
    let manifest: GitRuntimeManifest

    static func live() -> GitRuntimeConfiguration {
        let support = FileManager.default.urls(for: .applicationSupportDirectory, in: .userDomainMask)[0]
        return GitRuntimeConfiguration(
            applicationSupportDirectory: support,
            candidateSystemGitURLs: [
                URL(fileURLWithPath: "/opt/homebrew/bin/git"),
                URL(fileURLWithPath: "/usr/local/bin/git"),
                URL(fileURLWithPath: "/usr/bin/git")
            ],
            manifest: .current
        )
    }

    static func test(applicationSupportDirectory: URL, candidateSystemGitURLs: [URL], manifest: GitRuntimeManifest) -> GitRuntimeConfiguration {
        GitRuntimeConfiguration(
            applicationSupportDirectory: applicationSupportDirectory,
            candidateSystemGitURLs: candidateSystemGitURLs,
            manifest: manifest
        )
    }
}

actor GitRuntimeManager {
    static let shared = GitRuntimeManager()

    private let configuration: GitRuntimeConfiguration
    private let processRunner: GitRuntimeProcessRunning
    private let downloader: GitRuntimeDownloading
    private let extractor: GitRuntimeExtracting
    private var cachedSource: GitRuntimeSource?

    init(
        configuration: GitRuntimeConfiguration = .live(),
        processRunner: GitRuntimeProcessRunning = ProcessGitRuntimeRunner(),
        downloader: GitRuntimeDownloading = URLSessionGitRuntimeDownloader(),
        extractor: GitRuntimeExtracting = TarGitRuntimeExtractor()
    ) {
        self.configuration = configuration
        self.processRunner = processRunner
        self.downloader = downloader
        self.extractor = extractor
    }

    func availability() async -> GitRuntimeAvailability {
        if let source = await resolveExistingRuntime() {
            cachedSource = source
            return .ready(source)
        }
        return .missing
    }

    func executableURL() async throws -> URL {
        if let cachedSource {
            return cachedSource.executableURL
        }
        guard case .ready(let source) = await availability() else {
            throw GitRuntimeError.missing
        }
        return source.executableURL
    }

    private func resolveExistingRuntime() async -> GitRuntimeSource? {
        for url in configuration.candidateSystemGitURLs where FileManager.default.isExecutableFile(atPath: url.path) {
            if await validatesGit(at: url) {
                return .system(url)
            }
        }

        let managed = managedGitURL()
        if FileManager.default.isExecutableFile(atPath: managed.path), await validatesGit(at: managed) {
            return .managed(managed)
        }

        return nil
    }

    private func validatesGit(at url: URL) async -> Bool {
        guard let output = try? await processRunner.run(executableURL: url, arguments: ["--version"], currentDirectoryURL: nil) else {
            return false
        }
        return output.lowercased().contains("git version")
    }

    private func managedRootURL() -> URL {
        configuration.applicationSupportDirectory
            .appendingPathComponent("Commit+", isDirectory: true)
            .appendingPathComponent("Git", isDirectory: true)
            .appendingPathComponent(configuration.manifest.version, isDirectory: true)
    }

    private func managedGitURL() -> URL {
        managedRootURL().appendingPathComponent("bin/git")
    }
}

private extension GitRuntimeSource {
    var executableURL: URL {
        switch self {
        case .system(let url), .managed(let url):
            return url
        }
    }
}
```

```swift
final class ProcessGitRuntimeRunner: GitRuntimeProcessRunning {
    func run(executableURL: URL, arguments: [String], currentDirectoryURL: URL?) async throws -> String {
        let task = Process()
        task.executableURL = executableURL
        task.arguments = arguments
        task.currentDirectoryURL = currentDirectoryURL

        let stdout = Pipe()
        let stderr = Pipe()
        task.standardOutput = stdout
        task.standardError = stderr

        return try await withCheckedThrowingContinuation { continuation in
            task.terminationHandler = { process in
                let outData = stdout.fileHandleForReading.readDataToEndOfFile()
                let errData = stderr.fileHandleForReading.readDataToEndOfFile()
                let output = String(data: outData, encoding: .utf8) ?? ""
                let errorOutput = String(data: errData, encoding: .utf8) ?? ""

                if process.terminationStatus == 0 {
                    continuation.resume(returning: output)
                } else {
                    continuation.resume(throwing: GitRuntimeError.validationFailed(errorOutput.isEmpty ? executableURL.path : errorOutput))
                }
            }

            do {
                try task.run()
            } catch {
                continuation.resume(throwing: GitRuntimeError.validationFailed(executableURL.path))
            }
        }
    }
}

final class URLSessionGitRuntimeDownloader: GitRuntimeDownloading {
    func download(from url: URL) async throws -> Data {
        do {
            let (data, response) = try await URLSession.shared.data(from: url)
            guard let http = response as? HTTPURLResponse, (200..<300).contains(http.statusCode) else {
                throw GitRuntimeError.downloadFailed("Unexpected server response.")
            }
            return data
        } catch let error as GitRuntimeError {
            throw error
        } catch {
            throw GitRuntimeError.downloadFailed(error.localizedDescription)
        }
    }
}

final class TarGitRuntimeExtractor: GitRuntimeExtracting {
    func extract(archiveURL: URL, to destinationURL: URL) throws {
        try FileManager.default.createDirectory(at: destinationURL, withIntermediateDirectories: true)

        let task = Process()
        task.executableURL = URL(fileURLWithPath: "/usr/bin/tar")
        task.arguments = ["-xzf", archiveURL.path, "-C", destinationURL.path]

        try task.run()
        task.waitUntilExit()

        if task.terminationStatus != 0 {
            throw GitRuntimeError.extractionFailed("tar exited with status \(task.terminationStatus)")
        }
    }
}
```

```swift
func installManagedRuntime() async throws {
    let archiveData = try await downloader.download(from: configuration.manifest.url)
    let actualChecksum = sha256Hex(archiveData)
    guard actualChecksum == configuration.manifest.sha256 else {
        throw GitRuntimeError.checksumMismatch(expected: configuration.manifest.sha256, actual: actualChecksum)
    }

    let root = managedRootURL()
    let archiveURL = FileManager.default.temporaryDirectory
        .appendingPathComponent("commitplus-git-\(configuration.manifest.version).tar.gz")
    try archiveData.write(to: archiveURL, options: .atomic)

    try? FileManager.default.removeItem(at: root)
    try extractor.extract(archiveURL: archiveURL, to: root)
    try? FileManager.default.removeItem(at: archiveURL)

    let gitURL = managedGitURL()
    guard await validatesGit(at: gitURL) else {
        throw GitRuntimeError.validationFailed(gitURL.path)
    }

    cachedSource = .managed(gitURL)
}

private func sha256Hex(_ data: Data) -> String {
    SHA256.hash(data: data)
        .map { String(format: "%02x", $0) }
        .joined()
}
```

```swift
@MainActor
private func setGitRuntimeReady(_ ready: Bool, availability: GitRuntimeAvailability) {
    gitRuntimeReady = ready
    gitRuntimeAvailability = availability
}

private func prepareGitRuntime() async {
    let availability = await GitRuntimeManager.shared.availability()
    switch availability {
    case .ready:
        await setGitRuntimeReady(true, availability: availability)
    case .missing:
        await setGitRuntimeReady(false, availability: .installing(progress: nil))
        do {
            try await GitRuntimeManager.shared.installManagedRuntime()
            let updated = await GitRuntimeManager.shared.availability()
            await setGitRuntimeReady(true, availability: updated)
        } catch {
            await setGitRuntimeReady(false, availability: .failed(error.localizedDescription))

    }
}
```

```bash
#!/usr/bin/env bash
set -euo pipefail

if [[ $# -ne 3 ]]; then
  printf 'Usage: %s <runtime-root> <version> <output-dir>\n' "$0" >&2
  exit 64
fi

runtime_root="$1"
version="$2"
output_dir="$3"
platform="macos-universal"
archive_name="git-runtime-${version}-${platform}.tar.gz"
archive_path="${output_dir}/${archive_name}"
manifest_path="${output_dir}/git-runtime-${version}-${platform}.json"

  printf 'Expected executable Git at %s\n' "${runtime_root}/bin/git" >&2
  exit 66
fi

mkdir -p "$output_dir"
tar -czf "$archive_path" -C "$runtime_root" .
sha256="$(shasum -a 256 "$archive_path" | awk '{print $1}')"
size="$(stat -f '%z' "$archive_path")"
url="https://github.com/Tranthanh98/macgit/releases/download/git-runtime-${version}/${archive_name}"

cat > "$manifest_path" <<JSON
{
  "version": "${version}",
  "platform": "${platform}",
  "url": "${url}",
  "sha256": "${sha256}",
  "archiveSize": ${size}
}
JSON

cat > "$swift_manifest_path" <<SWIFT
import Foundation

enum GeneratedGitRuntimeManifest {
    static let current = GitRuntimeManifest(
        version: "${version}",
        platform: "${platform}",
        url: URL(string: "${url}")!,
        sha256: "${sha256}",
        archiveSize: ${size}
    )
}
SWIFT

printf 'Archive: %s\n' "$archive_path"
printf 'Manifest: %s\n' "$manifest_path"
printf 'Swift: %s\n' "$swift_manifest_path"
```
